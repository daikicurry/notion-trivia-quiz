async function startQuiz() { try { const response = await fetch('/.netlify/functions/getVocab'); const data = await response.json(); if (data.error) { console.log('API Error:', data.error); document.getElementById('quizDisplay').textContent = 'エラーが発生しました: ' + data.error; return; } if (!data.results || data.results.length === 0) { document.getElementById('quizDisplay').textContent = 'クイズデータが取得できませんでした'; document.getElementById('quizDisplay').innerHTML += '
デバッグ情報:
' + JSON.stringify(data.debug, null, 2); return; } // ランダムにクイズを選択 const randomIndex = Math.floor(Math.random() * data.results.length); currentQuiz = data.results[randomIndex]; // クイズを表示 document.getElementById('quizDisplay').textContent = currentQuiz.content; supplementShown = false; updateButtons('quiz'); } catch (error) { console.error('Error:', error); document.getElementById('quizDisplay').textContent = 'エラーが発生しました: ' + error.message; } }
